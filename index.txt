    <script>
        document.addEventListener('DOMContentLoaded', () => {
            
            feather.replace();
            initMobileMenu();
            initAnalysisSearch();
            initNavigation();

            // --- LISTA DE ACCIONES COMPARTIDA ---
            const stockList = [
                { symbol: 'BCS:FALABELLA', name: 'Falabella SA' }, { symbol: 'BCS:SQM-B', name: 'SQM' }, { symbol: 'BCS:COPEC', name: 'Empresas Copec SA' },
                { symbol: 'AAPL', name: 'Apple Inc.' }, { symbol: 'GOOGL', name: 'Alphabet Inc.' }, { symbol: 'MSFT', name: 'Microsoft' },
                { symbol: 'SP_IPSA', name: 'S&P IPSA Chile' }, { symbol: 'FOREXCOM:SPXUSD', name: 'S&P 500' }, { symbol: 'BITSTAMP:BTCUSD', name: 'Bitcoin / USD' }
            ];

            function initMobileMenu() {
                const desktopNav = document.getElementById('desktop-nav');
                const mobileMenu = document.getElementById('mobile-menu');
                mobileMenu.innerHTML = desktopNav.innerHTML;
                mobileMenu.querySelectorAll('a').forEach(link => {
                    link.classList.remove('flex', 'items-center');
                    link.classList.add('block', 'px-4', 'py-3', 'hover:bg-gray-700');
                    link.querySelector('i')?.remove();
                });

                document.getElementById('mobile-menu-button').addEventListener('click', () => {
                    mobileMenu.classList.toggle('hidden');
                });
            }

            // --- WIDGET DE GRÁFICO AVANZADO ---
            function createAdvancedChartWidget(ticker) {
                const container = document.getElementById('advanced-chart-container');
                if (!container) return;
                container.innerHTML = '';
                new TradingView.widget({
                  "autosize": true,
                  "symbol": ticker,
                  "interval": "D",
                  "timezone": "America/Santiago",
                  "theme": "dark",
                  "style": "1",
                  "locale": "es",
                  "details": true,
                  "allow_symbol_change": true,
                  "calendar": false,
                  "hide_side_toolbar": false,
                  "withdateranges": true,
                  "studies": [
                    "STD;SMA",
                    "STD;RSI"
                  ],
                  "backgroundColor": "#161B22",
                  "gridColor": "rgba(242, 242, 242, 0.05)",
                  "container_id": "advanced-chart-container"
                });
            }

            // --- LÓGICA DE BÚSQUEDA ---
            function setupSearch(inputId, suggestionsId, buttonId, onSearchCallback) {
                const tickerInput = document.getElementById(inputId);
                const suggestionsBox = document.getElementById(suggestionsId);
                const searchButton = document.getElementById(buttonId);

                const performSearch = () => {
                    const ticker = tickerInput.value.trim().toUpperCase();
                    if (ticker) {
                        onSearchCallback(ticker);
                        suggestionsBox.classList.add('hidden');
                        tickerInput.blur();
                    }
                };

                tickerInput.addEventListener('input', () => {
                    const inputText = tickerInput.value.trim().toUpperCase();
                    suggestionsBox.innerHTML = '';
                    if (!inputText) { suggestionsBox.classList.add('hidden'); return; }

                    const filtered = stockList.filter(s => s.symbol.toUpperCase().includes(inputText) || s.name.toUpperCase().includes(inputText)).slice(0, 5);
                    
                    if (filtered.length > 0) {
                        suggestionsBox.classList.remove('hidden');
                        filtered.forEach(stock => {
                            const item = document.createElement('div');
                            item.classList.add('suggestion-item');
                            item.innerHTML = `<strong>${stock.symbol}</strong><div class="suggestion-name">${stock.name}</div>`;
                            item.addEventListener('click', () => {
                                tickerInput.value = stock.symbol;
                                performSearch();
                            });
                            suggestionsBox.appendChild(item);
                        });
                    } else {
                        suggestionsBox.classList.add('hidden');
                    }
                });

                document.addEventListener('click', (e) => {
                    if (!e.target.closest('.relative')) suggestionsBox.classList.add('hidden');
                });

                searchButton.addEventListener('click', performSearch);
                tickerInput.addEventListener('keypress', (e) => { if (e.key === 'Enter') performSearch(); });
            }

            function initAnalysisSearch() {
                setupSearch('ticker-input-analysis', 'suggestions-box-analysis', 'search-button-analysis', createAdvancedChartWidget);
            }

            function initNavigation() {
                const navLinks = document.querySelectorAll('.nav-link');
                const sectionButtons = document.querySelectorAll('.section-button');
                const backButtons = document.querySelectorAll('.back-button');
                const logoLink = document.getElementById('logo-link');
                const tabSelection = document.getElementById('tab-selection');
                const sectionsContainer = document.getElementById('sections-container');
                const pageSections = document.querySelectorAll('.page-section');
                const widgetLoadStatus = { 'criptomonedas-section': false, 'analisis-section': false };

                const loadWidget = (targetId) => {
                    if (targetId === 'criptomonedas-section' && !widgetLoadStatus[targetId]) {
                        new TradingView.widget({ "dataSource": "Crypto", "blockSize": "market_cap_calc", "blockColor": "24h_close_change|5", "locale": "es", "colorTheme": "dark", "isTransparent": true, "width": "100%", "height": "100%", "container_id": "crypto-heatmap-container" });
                        widgetLoadStatus[targetId] = true;
                    }
                    if (targetId === 'analisis-section' && !widgetLoadStatus[targetId]) {
                        createAdvancedChartWidget('BCS:SP_IPSA');
                        widgetLoadStatus[targetId] = true;
                    }
                };
                
                const showSection = (targetId) => {
                    tabSelection.classList.add('hidden');
                    sectionsContainer.classList.remove('hidden');
                    pageSections.forEach(section => section.classList.add('hidden'));
                    
                    const targetSection = document.getElementById(targetId);
                    if (targetSection) targetSection.classList.remove('hidden');

                    loadWidget(targetId);

                    navLinks.forEach(link => link.classList.toggle('active', link.dataset.target === targetId));
                    window.scrollTo(0, 0);
                };

                const showTabSelection = () => {
                    tabSelection.classList.remove('hidden');
                    sectionsContainer.classList.add('hidden');
                    navLinks.forEach(link => link.classList.remove('active'));
                };

                document.querySelectorAll('.nav-link, .section-button').forEach(el => {
                    el.addEventListener('click', (e) => {
                        e.preventDefault();
                        const targetId = e.currentTarget.dataset.target;
                        if (targetId) {
                            showSection(targetId);
                            document.getElementById('mobile-menu').classList.add('hidden');
                        }
                    });
                });

                backButtons.forEach(button => button.addEventListener('click', showTabSelection));
                logoLink.addEventListener('click', (e) => { e.preventDefault(); showTabSelection(); });

                showTabSelection();
            }
        });
    </script>
